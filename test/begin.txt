Private Sub TextBox1_Change()
Dim NotNumeric As Boolean

Dim TempValue As String

If Not IsNumeric(TextBox1.Value) Then

    If TextBox1.Value <> "" Then
    
          NotNumeric = True
          
          
          TempValue = Left(Me.TextBox1.Value, Len(Me.TextBox1.Value) - 1)
          
          While NotNumeric = True And TempValue <> ""
          
               If Not IsNumeric(TempValue) Then
          
                    TempValue = Left(TempValue, Len(TempValue) - 1)
                    
               
               Else
                    
                    NotNumeric = False
                    
               End If
    
          Wend
          
          Me.TextBox1.Value = TempValue
            
    End If
    
 End If
If TextBox1.Value <> "" Then
    If TextBox1.Value > 90 Then
        TextBox1 = Left(TextBox1, Len(TextBox1) - 1)
     End If
End If

If TextBox1.Value <> "" Then
    If Right(TextBox1.Value, 1) = " " Or Right(TextBox1.Value, 1) = "+" Then
        TextBox1 = Left(TextBox1, Len(TextBox1) - 1)
        End If
        
End If





End Sub
